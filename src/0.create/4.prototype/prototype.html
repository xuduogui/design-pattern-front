            document.getElementById('prototype-button')
              .addEventListener('click', client);

            function client() {
              const dom = document.getElementById('prototype-text');
              dom.innerHTML = '';
              const concretePrototype = new ConcretePrototype();
              let max = 10;
              let index = 0;
              while (index < max) {
                const clonePrototype = concretePrototype.clone();
                clonePrototype.setPrototypeName(`
我是克隆的第${index + 1}个原型数据
`)
                dom.innerHTML += clonePrototype.getPrototypeName();
                index++;
              };
            };

            function AbstractPrototype() {
              this.prototypeName = '';
              this.prototypePart = '';
              this.setPrototypeName = function (value) {};
              this.getPrototypeName = function (value) {};
              this.setPrototypePart = function (value) {};
              this.getPrototypePart = function (value) {};
              this.clone = function () {};
            };

            function ConcretePrototype() {
              this.prototypeName = 'prototype'
              this.prototypePart = 'prototypePart'
              this.setPrototypeName = function (value) {
                this.prototypeName = value;
              };
              this.getPrototypeName = function (value) {
                return this.prototypeName;
              };
              this.setPrototypePart = function (value) {
                this.prototypePart = value;
              };
              this.getPrototypePart = function (value) {
                return this.prototypePart;
              };
              this.clone = function () {
                return Object.create(this);
              };
            };
            ConcretePrototype.prototype = new AbstractPrototype()

            // ==== 原型模式 END ====
